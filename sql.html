                        SELECT DATE ( a.data_alteracao ) as "Data Alteração", 
                        SUM(a.valor_liberado) as "Soma de Valor Liberado", 
                        COUNT ((case when sa.efetivo = 't' then sa.efetivo else null end)) AS "Efetivo",
                        COUNT ((case when sa.alo = 't' then sa.alo else null end)) AS "CPC",
                        COUNT ((case when sa.target = 't' then sa.target else null end)) AS "Target",
                        COUNT ((case when sa.convercao = 't' then sa.convercao else null end)) AS "Conversão",
                        COUNT ((case when sa.sem_interesse ='t' then sa.sem_interesse else null end)) AS "Sem Interesse", 
                        COUNT ( ( CASE WHEN A.status = 46 THEN A.status ELSE NULL END ) ) AS "Sem Perfil"
                        FROM atendimento a
                        JOIN campanha c ON c."id" = a.campanha
                        JOIN status_atendimento sa ON sa."id" = a.status
                        JOIN usuario u ON u."id" = a.usuario_alteracao
                        JOIN equipe e ON e."id" = u.equipe 
                        WHERE
                        c.descricao like '%PROATIVA_CARTÃO_3004_Fila_02%' or c.descricao like '%PROATIVA_CARTÃO_3004_Fila_01%'
                        GROUP BY "Data Alteração"
                        ORDER BY "Data Alteração" ASC
